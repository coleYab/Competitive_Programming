# Problem: Count Primes - https://leetcode.com/problems/count-primes/

class Solution {
public:
    int countPrimes(int n) {
        vector<bool> seive(n + 10, true);
        for (int i = 2; i <= n; i++) {
            if (!seive[i]) continue;
            for (int j = i + i; j <= n; j+=i) {
                seive[j] = false;
            }
        }

        int ans = 0;
        for (int i = 2; i < n; i++) {
            ans += seive[i];
        }

        return ans;
    }
};
