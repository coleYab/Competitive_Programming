# Problem: Divisibility by 2^n - https://codeforces.com/contest/1744/problem/D

#include <bits/stdc++.h>

using namespace std;

#define multi
#define int long long

struct test {
    void solve() {
        int n; cin >> n;
        vector<int> a(n), k(n + 1, 0);
        for (auto &it : a) cin >> it;
        int cnt = 0;
        for (auto it : a) {
            int c = 0;
            while (it % (1LL << c) == 0) {
                c++;
            }
            cnt += (c - 1);
        }

        int sm = cnt;
        for (int i = 1; i <= n; i++) {
            int c = 0;
            while (i % (1LL << c) == 0) c++; 
            k[i] = c - 1;
            sm += k[i];
        }

        if (sm < n) {
            cout << -1 << endl;
            return;
        }
        sort(k.rbegin(), k.rend());
        int ans = 0;
        int i = 0;
        while (cnt < n) {
            cnt += k[i];
            ans++;
            i++;
        }

        cout << ans << endl;
    }
};

int32_t main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int tt = 1;
#ifdef multi
    cin >> tt;
#endif
    while (tt--) {
        test().solve();
    }
}
