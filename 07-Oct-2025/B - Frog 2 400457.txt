# Problem: B - Frog 2 - https://atcoder.jp/contests/dp/tasks/dp_b

#include <bits/stdc++.h>

using namespace std;

// #define multi
#define int long long

struct test {
    void solve() {
        int n, m; cin >> n >> m;
        vector<int> a(n);
        for (auto &it : a) cin >> it;
        vector<int> dp(n + 1, 1e9);
        dp[0] = 0;            
        for (int i = 0; i < n; i++) {
            for (int j = i - 1; j > i - 1 - m; j--) {
                if (j >= 0) {
                    dp[i] = min(dp[i], dp[j] + abs(a[i] - a[j]));
                }
            }
        }

        cout << dp[n - 1] << endl;
    }
};

int32_t main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int tt = 1;
#ifdef multi
    cin >> tt;
#endif
    while (tt--) {
        test().solve();
    }
}
