struct test {
    void solve() {
        int n, m; cin >> n >> m;
        int two = n / 2, one = n % 2;
        int tot = two + one;
        while (tot % m != 0) {
            if (!two) break;
            two -= 1;
            one += 2;
            tot = two + one;
        }

        cout << (tot % m ? -1 : tot) << endl;   
    }
};

    int tt = 1;
    // cin >> tt;
    while (tt--) {
        test().solve();
    }
