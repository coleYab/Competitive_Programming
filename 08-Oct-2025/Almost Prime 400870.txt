# Problem: Almost Prime - https://codeforces.com/problemset/problem/26/A

#include <bits/stdc++.h>
#include <vector>
using namespace std;

#define ll long long
#define int long long

void solve() {
    int n; cin >> n;
    vector<int> arr(n + 1, 0);
    int ans = 0;
    arr[0] = 1;
    arr[1] = 1;
    for (int i = 0; i <= n; i++) {
        if (arr[i] == 0) {
            for (int j = i; j <= n; j+=i) {
                arr[j] += 1;
            }
        }

        if (arr[i] == 2) ans++;
    }

    cout << ans << "\n";
}

int32_t main()
{
    int tc = 1;
    // cin >> tc;

	while(tc--){
		solve();
	}

	return 0;
}
